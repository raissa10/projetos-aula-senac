SELECT
	cinema.tbfilme.filcodigo,
	cinema.tbfilme.filtitulo,
	cinema.tbfilme.fildataestreia,
	cinema.tbcategoria.catnome,
	cinema.tbator.atonome,
	cinema.tbfilme.filsinopse,
	cinema.tbfilme.filtrailer
FROM
	cinema.tbatorfilme
	INNER JOIN cinema.tbfilme on 
	cinema.tbatorfilme.filcodigo=cinema.tbfilme.filcodigo
	INNER JOIN cinema.tbator on 
	cinema.tbator.atocodigo=cinema.tbatorfilme.atocodigo
	INNER JOIN cinema.tbcategoria on
	cinema.tbcategoria.catcodigo=cinema.tbfilme.catcodigo
ORDER BY 
	1

    public function getNomeAtorFilme($CodigoFilme) {
        $aAtores = Array();
        $sSql = "SELECT";
        $sSql.= "     cinema.tbator.atonome";
        $sSql.= " FROM";
        $sSql.= "     cinema.tbatorfilme";
        $sSql.= "     INNER JOIN cinema.tbator on";
        $sSql.= "     cinema.tbator.atocodigo=cinema.tbatorfilme.atocodigo";
        $sSql.= " WHERE";
        $sSql.= "     cinema.tbatorfilme.filcodigo=" . $CodigoFilme;

        $aAtorBanco = $this->getQuery()->selectAll($sSql);
        foreach ($aAtorBanco as $aValor) {
            $oModelAtor = $this->getModelFromDb($aValor);
            $aAtores [] = $oModelAtor;
        }
        return $aAtores;
    }
